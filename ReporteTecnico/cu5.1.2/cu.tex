\begin{UseCase}{CU 5.1.2}{Gestionar precondiciones}
	{
		Este caso de uso permite al analista visualizar las precondiciones del caso de uso. También permite 
		al actor acceder a las operaciones de registro, modificación y eliminación de las precondiciones.
	}
	\UCitem{Actor}{\cdtRef{actor:liderAnalisis}{Líder de análisis}, \cdtRef{actor:analista}{Analista}}
	\UCitem{Propósito}{
		Revisar y gestionar las precondiciones de un caso de uso.
	}
	\UCitem{Entradas}{
		Ninguna
	}
	\UCitem{Salidas}{
		\begin{UClist}
			\UCli \cdtRef{Precondicion}{Precondición}: \ioTabla{el \cdtRef{Precondicion:Numero}{Número} y la \cdtRef{Precondicion:Redacción}{Redacción}}.
		\end{UClist}
	}
	
	\UCitem{Mensajes}{
		\begin{UClist}
			\UCli \cdtIdRef{MSG2}{No existe información}: Se muestra en la pantalla \cdtIdRef{IU 5.1}{Registrar caso de uso} o \cdtIdRef{IU 5.2}{Modificar caso de uso} cuando no existen precondiciones registradas.
		\end{UClist}
	}

	\UCitem{Precondiciones}{
		Ninguna
	}
	
	\UCitem{Postcondiciones}{
		\begin{UClist}
			\UCli Se podrá solicitar el registro de una precondición por medio del caso de uso \cdtIdRef{CU 5.1.2.1}{Registrar precondición}.
			\UCli Se podrá solicitar la modificación de una precondición por medio del caso de uso \cdtIdRef{CU 5.1.2.2}{Modificar precondición}.
			\UCli Se podrá solicitar la eliminación de una precondición por medio del caso de uso \cdtIdRef{CU 5.1.2.3}{Eliminar precondición}.
		\end{UClist}
	}

	\UCitem{Errores}{
		Ninguno
	}

	\UCitem{Tipo}{
		Secundario, extiende del caso de uso \cdtIdRef{CU 5.1}{Registrar caso de uso}.
	}
\end{UseCase}
%-------------------------------------------------------%trayectoria Principal-----------------------------------------------
 \begin{UCtrayectoria}
    \UCpaso[\UCactor] Solicita gestionar las precondiciones del caso de uso seleccionando la opción ``Precondiciones'' de la pantalla \cdtIdRef{IU 5.1}{Registrar caso de uso} o \cdtIdRef{IU 5.2}{Modificar caso de uso}.
    \UCpaso[\UCsist] Busca información de las precondiciones del caso de uso. \refTray{A}
    \UCpaso[\UCsist] Muestra la información de las precondiciones en la pantalla \cdtIdRef{IU 5.1}{Registrar caso de uso} o \cdtIdRef{IU 5.2}{Modificar caso de uso}. 
    \UCpaso[\UCactor] Gestiona las precondiciones a través de los botones: \btnAgregar, \btnEditar y \btnEliminar. \label{cu5.1.2:gestionaCU}
 \end{UCtrayectoria}
 
 \begin{UCtrayectoriaA}[Fin del caso de uso]{A}{No existen registros de precondiciones.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG2}{No existe información} en pantalla correspondiente
    para indicar que no hay registros de precondiciones para mostrar.
 \end{UCtrayectoriaA}
