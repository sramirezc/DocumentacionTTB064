\begin{UseCase}{CU 5.1.4}{Gestionar puntos de extensión}
	{
		Este caso de uso permite al analista visualizar los puntos de extensión del caso de uso. También permite 
		al actor acceder a las operaciones de registro, modificación y eliminación de las puntos de extensión.
	}

	\UCitem{Actor}{\cdtRef{actor:liderAnalisis}{Líder de análisis}, \cdtRef{actor:analista}{Analista}}
	\UCitem{Propósito}{
		Revisar y gestionar los puntos de extensión de un caso de uso.
	}
	\UCitem{Entradas}{
		Ninguna
	}
	\UCitem{Salidas}{
		\begin{UClist}
			\UCli \cdtRef{Extension}{Punto de Extensión}: \ioTabla{la \cdtRef{Extension:Causa}{Causa}, la región de la trayectoria y el caso de uso que extiende}.
		\end{UClist}
	}
	
	\UCitem{Mensajes}{
		\begin{UClist}
			\UCli \cdtIdRef{MSG2}{No existe información}: Se muestra en la pantalla \cdtIdRef{IU 5.1}{Registrar caso de uso} o \cdtIdRef{IU 5.2}{Modificar caso de uso} cuando no existen puntos de extensión registradas.
		\end{UClist}
	}

	\UCitem{Postcondiciones}{
		Ninguna
	}
	
	\UCitem{Postcondiciones}{
		\begin{UClist}
			\UCli Se podrá solicitar el registro de un punto de extensión por medio del caso de uso \cdtIdRef{CU 5.1.4.1}{Registrar punto de extensión}.
			\UCli Se podrá solicitar la modificación de un punto de extensión por medio del caso de uso \cdtIdRef{CU 5.1.4.2}{Modificar punto de extensión}.
			\UCli Se podrá solicitar la eliminación de un punto de extensión por medio del caso de uso \cdtIdRef{CU 5.1.4.3}{Eliminar punto de extensión}.
		\end{UClist}
	}

	\UCitem{Errores}{
		Ninguno
	}

	\UCitem{Tipo}{
		Secundario, extiende del caso de uso \cdtIdRef{CU 5.1}{Registrar caso de uso}.
	}
\end{UseCase}
%-------------------------------------------------------%trayectoria Principal-----------------------------------------------
 \begin{UCtrayectoria}
    \UCpaso[\UCactor] Solicita gestionar los puntos de extensión del caso de uso seleccionando la opción ``Puntos de Extensión'' de la pantalla \cdtIdRef{IU 5.1}{Registrar caso de uso}.
    \UCpaso[\UCsist] Busca información de los puntos de extensión del caso de uso. \refTray{A}
    \UCpaso[\UCsist] Muestra la información de los puntos de extensión en la pantalla \cdtIdRef{IU 5.1}{Registrar caso de uso} o \cdtIdRef{IU 5.2}{Modificar caso de uso}. 
    \UCpaso[\UCactor] Gestiona los puntos de extensión a través de los botones: \btnAgregar, \btnEditar y \btnEliminar. \label{cu5.1.4:gestionaCU}
 \end{UCtrayectoria}
 
 \begin{UCtrayectoriaA}[Fin del caso de uso]{A}{No existen registros de puntos de extensión.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG2}{No existe información} en la pantalla correspondiente
    para indicar que no hay registros de puntos de extensión para mostrar.
 \end{UCtrayectoriaA}
