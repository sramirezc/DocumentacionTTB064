\begin{UseCase}{CU 8}{Gestionar reglas de negocio}
	{
		Este caso de uso permite al analista visualizar los registros de las reglas de negocio del sistema. También permite 
		al actor acceder a las operaciones de registro, consulta, modificación y eliminación de una regla de negocio.
	}
	
	\UCitem{Actor}{\cdtRef{actor:liderAnalisis}{Líder de análisis}, \cdtRef{actor:analista}{Analista}}
	\UCitem{Propósito}{
		Gestionar las reglas de negocio registradas en la herramienta.
	}
	\UCitem{Entradas}{
		Ninguna
	}
	\UCitem{Salidas}{
		\begin{UClist}
			\UCli \cdtRef{}{Clave} y \cdtRef{}{nombre} del proyecto.
			\UCli \cdtRef{ReglaDeNegocio}{Regla de Negocio}: \ioTabla{el \cdtRef{Elemento:Nombre}{Nombre}}{de las reglas de negocio}.
		\end{UClist}
	}
	
	\UCitem{Mensajes}{
		\begin{UClist}
			\UCli \cdtIdRef{MSG2}{No existe información}: Se muestra en la pantalla \cdtIdRef{CU 8}{Gestionar reglas de negocio} cuando no existen reglas de negocio registradas.
		\end{UClist}
	}

	\UCitem{Precondiciones}{
		Ninguna
	}
	
	\UCitem{Postcondiciones}{
		\begin{UClist}
			\UCli Se podrá solicitar el registro de una regla de negocio por medio del caso de uso \cdtIdRef{CU 8.1}{Registrar regla de negocio}.
			\UCli Se podrá solicitar la modificación de una regla de negocio por medio del caso de uso \cdtIdRef{CU 8.2}{Modificar regla de negocio}.
			\UCli Se podrá solicitar la eliminación de una regla de negocio por medio del caso de uso \cdtIdRef{CU 8.3}{Eliminar regla de negocio}.
			\UCli Se podrá solicitar la consulta de una regla de negocio por medio del caso de uso \cdtIdRef{CU 8.4}{Consultar regla de negocio}.
		\end{UClist}
	}

	\UCitem{Errores}{
		Ninguno
	}

	\UCitem{Tipo}{
		Primario
	}
\end{UseCase}
%-------------------------------------------------------%trayectoria Principal-----------------------------------------------
 \begin{UCtrayectoria}
    \UCpaso[\UCactor] Solicita gestionar las reglas de negocio seleccionando la opción ``Reglas de negocio'' del \cdtRef{menu:principal}{Menú principal}.
    \UCpaso[\UCsist] Busca la información de las reglas de negocio registradas en el sistema. \refTray{A}
    \UCpaso[\UCsist] Muestra la información de las reglas de negocio en la pantalla \cdtIdRef{IU 8}{Gestionar reglas de negocio} y las operaciones 
    disponibles de acuerdo a la regla de negocio \cdtIdRef{RN15}{Elementos en uso}. 
    \UCpaso[\UCactor] Gestiona las reglas de negocio a través de los botones: \cdtButton{Registrar}, \btnConsulta, \btnEditar y \btnEliminar. \label{cu8:gestiona}
 \end{UCtrayectoria}
 
 \begin{UCtrayectoriaA}[Fin del caso de uso]{A}{No existen registros de reglas de negocio.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG2}{No existe información} en pantalla \cdtIdRef{IU 8}{Gestionar reglas de negocio} 
    para indicar que no hay registros de reglas de negocio para mostrar.
 \end{UCtrayectoriaA}
